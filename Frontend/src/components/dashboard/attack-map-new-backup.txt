// Types
interface AttackData {

  sourceLng: number;
  sourceCountry: string;
  targetIp: string;
  targetLat: number;
  targetLng: number;

interface ThreatData {

  size: number;

}

interface URLhausResponse {
  url: string;
  threat?: string;
  date_added?: string;
  url_status?: string;
  tags?: string[];
}

// Real threat intelligence from URLhaus API
const fetchURLhausThreatData = async (): Promise<ThreatData[]> => {
  try {
    // Fetch recent URLs from URLhaus API
    const response = await fetch('https://urlhaus-api.abuse.ch/v1/urls/recent/', {
      method: 'GET',
      headers: {
        'User-Agent': 'SOC-Dashboard'
      }
    });

    if (!response.ok) {
      throw new Error('URLhaus API request failed');
    }

    const data = await response.json();
    
    if (data.query_status === 'ok' && data.urls) {
      // Sample locations for demonstration (in real implementation, you'd use IP geolocation)
      const sampleLocations = [
        { lat: 39.9042, lng: 116.4074, country: 'China' },
        { lat: 55.7558, lng: 37.6173, country: 'Russia' },
        { lat: 28.6139, lng: 77.2090, country: 'India' },
        { lat: -23.5505, lng: -46.6333, country: 'Brazil' },
        { lat: 52.5200, lng: 13.4050, country: 'Germany' },
        { lat: 25.2048, lng: 55.2708, country: 'UAE' },
        { lat: 40.7128, lng: -74.0060, country: 'USA' },
        { lat: 51.5074, lng: -0.1278, country: 'UK' },
        { lat: 35.6762, lng: 139.6503, country: 'Japan' },
        { lat: 48.8566, lng: 2.3522, country: 'France' }
      ];

      return data.urls.slice(0, 100).map((urlData: any, index: number) => {
        const location = sampleLocations[index % sampleLocations.length];
        const threat = urlData.threat || 'malware';
        const severityColors = {
          'malware_download': '#ff4444',
          'malware': '#ff6666',
          'exploit_kit': '#ff8800',
          'phishing': '#ffaa00',
          'default': '#ffdd00'
        };

        return {
          lat: location.lat + (Math.random() - 0.5) * 10,
          lng: location.lng + (Math.random() - 0.5) * 10,
          size: Math.random() * 0.8 + 0.2,
          color: severityColors[threat as keyof typeof severityColors] || severityColors.default,
          attackType: threat,
          count: Math.floor(Math.random() * 500) + 50,
          country: location.country
        };
      });
    }
  } catch (error) {
    console.error('Failed to fetch URLhaus data:', error);
  }

  // Fallback to mock data if API fails
  return generateMockThreatData();
};

// Enhanced 3D Globe Component with auto-rotation and real threat data
const Globe3DMap: React.FC<{ threats: ThreatData[] }> = ({ threats }) => {
  const globeRef = useRef<any>();
  const [autoRotate, setAutoRotate] = useState(true);
  const lastInteractionTime = useRef(Date.now());

  useEffect(() => {
    if (globeRef.current) {
      const globe = globeRef.current;
      
      // Enhanced 3D globe settings for better quality
      globe.controls().autoRotate = autoRotate;
      globe.controls().autoRotateSpeed = 1.2;
      globe.controls().enableDamping = true;
      globe.controls().dampingFactor = 0.05;
      globe.controls().minDistance = 200;
      globe.controls().maxDistance = 800;
      
      // Better rendering quality
      globe.renderer().setPixelRatio(Math.min(window.devicePixelRatio, 2));
      globe.renderer().antialias = true;
      globe.renderer().shadowMap.enabled = true;

      // Handle user interaction
      const handleInteractionStart = () => {
        lastInteractionTime.current = Date.now();
        if (autoRotate) {
          setAutoRotate(false);
        }
      };

      const handleInteractionEnd = () => {
        lastInteractionTime.current = Date.now();
      };

      // Listen for interaction events
      const controls = globe.controls();
      controls.addEventListener('start', handleInteractionStart);
      controls.addEventListener('end', handleInteractionEnd);

      // Resume auto-rotation after 8 seconds of inactivity
      const checkInactivity = setInterval(() => {
        if (Date.now() - lastInteractionTime.current > 8000 && !autoRotate) {
          setAutoRotate(true);
        }
      }, 1000);

      return () => {
        clearInterval(checkInactivity);
        if (controls) {
          controls.removeEventListener('start', handleInteractionStart);
          controls.removeEventListener('end', handleInteractionEnd);
        }
      };
    }
  }, [autoRotate]);

  useEffect(() => {
    if (globeRef.current) {
      globeRef.current.controls().autoRotate = autoRotate;
    }
  }, [autoRotate]);

  return (
    <div className="flex flex-col h-full">
      <div className="mb-4">
        <h3 className="text-lg font-semibold text-white mb-2">Global Threat Intelligence</h3>
        <div className="space-y-1 text-sm">
          <div className="flex items-center gap-2">
            <div className="w-3 h-3 rounded-full bg-red-500"></div>
            <span className="text-gray-300">Malware Distribution</span>
          </div>
          <div className="flex items-center gap-2">
            <div className="w-3 h-3 rounded-full bg-orange-500"></div>
            <span className="text-gray-300">Phishing/Exploits</span>
          </div>
          <div className="flex items-center gap-2">
            <div className="w-3 h-3 rounded-full bg-yellow-500"></div>
            <span className="text-gray-300">Suspicious Activity</span>
          </div>
        </div>
      </div>
      <div className="flex-1 bg-black rounded-lg overflow-hidden border border-gray-700 relative">
        <Globe3D
          ref={globeRef}
          globeImageUrl="//unpkg.com/three-globe/example/img/earth-blue-marble.jpg"
          bumpImageUrl="//unpkg.com/three-globe/example/img/earth-topology.png"
          backgroundImageUrl="//unpkg.com/three-globe/example/img/night-sky.png"
          showAtmosphere={true}
          atmosphereColor="#4f46e5"
          atmosphereAltitude={0.2}
          pointsData={threats}
          pointAltitude={0.02}
          pointColor="color"
          pointRadius={(d: any) => d.size * 2.5}
          pointResolution={8}
          pointLabel={(d: any) => `
            <div class="bg-gray-900/95 p-3 rounded-lg shadow-xl border border-gray-600 backdrop-blur-sm">
              <div class="text-white font-semibold text-sm">${d.attackType}</div>
              <div class="text-gray-300 text-xs mt-1">Threats: ${d.count}</div>
              ${d.country ? `<div class="text-gray-400 text-xs">Origin: ${d.country}</div>` : ''}
            </div>
          `}
          width={400}
          height={500}
          enablePointerInteraction={true}
          animateIn={true}
        />
        {autoRotate && (
          <div className="absolute top-2 right-2 bg-blue-600/80 text-white px-2 py-1 rounded text-xs">
            Auto-rotating
          </div>
        )}
      </div>
    </div>
  );
};

// Main Component
export function AttackMap() {
  const [attacks, setAttacks] = useState<AttackData[]>([]);
  const [threats, setThreats] = useState<ThreatData[]>([]);
  const [isLoading, setIsLoading] = useState(true);

  // Load initial data
  useEffect(() => {
    const loadData = async () => {
      setIsLoading(true);
      try {
        const [attackData, threatData] = await Promise.all([
          Promise.resolve(generateMockAttackData()),
          fetchURLhausThreatData()
        ]);
        setAttacks(attackData);
        setThreats(threatData);
      } catch (error) {
        console.error('Error loading data:', error);
        // Fallback to mock data
        setAttacks(generateMockAttackData());
        setThreats(generateMockThreatData());
      } finally {
        setIsLoading(false);
      }
    };

    loadData();
  }, []);

  // Auto-refresh data every 30 seconds
  useEffect(() => {
    const interval = setInterval(async () => {
      try {
        const [newAttacks, newThreats] = await Promise.all([
          Promise.resolve(generateMockAttackData()),
          fetchURLhausThreatData()
        ]);
        setAttacks(newAttacks);
        setThreats(newThreats);
      } catch (error) {
        console.error('Error refreshing data:', error);
      }
    }, 30000);

    return () => clearInterval(interval);
  }, []);

  if (isLoading) {
    return (
      <div className="w-full h-[600px] bg-gray-800 rounded-lg p-6 border border-gray-700 flex items-center justify-center">
        <div className="text-center">
          <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-blue-500 mx-auto"></div>
          <p className="text-gray-300 mt-4">Loading threat intelligence...</p>
        </div>
      </div>
    );
  }

  return (
    <div className="w-full h-[600px] bg-gray-800 rounded-lg p-6 border border-gray-700">
      <div className="flex gap-6 h-full">
        {/* 2D Map - 2/3 width */}
        <div className="flex-1" style={{ flexBasis: '66.666%' }}>
          <Map2D attacks={attacks} />
        </div>
        
        {/* 3D Globe - 1/3 width */}
        <div className="flex-1" style={{ flexBasis: '33.333%' }}>
          <Globe3DMap threats={threats} />
        </div>
      </div>
    </div>
  );
}
